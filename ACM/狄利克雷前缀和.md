##### 狄利克雷前缀和

$$
f*1=g，则我们称g是f的狄利克雷前缀和
$$

```cpp
//给定f1,f2,...,fn，求g函数，O(nlognlogn)
for (int i = 1; i <= cnt; ++ i)
  for (int j = 1; j * Prime[i] <= n; ++ j) f[j * Prime[i]] += f[j];
```



### 模板

$$
给定a_1,a_2,...,a_n，求b序列满足b_k=\sum_{i|k}a_i
$$

```cpp
//luoguP5495 Dirichlet 前缀和
#include<bits/stdc++.h>
using namespace std;
const int N=2e7+7;

#define uint unsigned int
uint seed;
inline uint getnext(){
	seed^=seed<<13;
	seed^=seed>>17;
	seed^=seed<<5;
	return seed;
}

uint a[N],ans;
int n,cnt,np[N],pri[N];

void solve(){
	np[1]=1;
	for(int i=2;i<=n;i++){
		if(!np[i]) pri[++cnt]=i;
		for(int j=1;j<=cnt&&pri[j]*i<=n;j++){
			np[i*pri[j]]=1;
			if(i%pri[j]==0) break;
		}
	} 
}

signed main(){
	cin>>n>>seed;
	solve();
	for(int i=1;i<=n;i++) a[i]=getnext();
	for(int i=1;i<=cnt;i++){
		for(int j=1;j*pri[i]<=(uint)n;j++) a[pri[i]*j]+=a[j];
	}
	for(int i=1;i<=n;i++) ans^=a[i];
	printf("%u\n",ans);
	return 0;
}

```
